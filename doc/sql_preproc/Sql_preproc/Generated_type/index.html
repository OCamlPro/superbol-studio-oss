type ref_value =
  1. | Reference of {
    1. prefix : string;
    2. var : string;
    }
  2. | Value of {
    1. prefix : string;
    2. var : string;
    }
type error_treatment = {
  1. prefix : string;
  2. not_found_whenever : whenever_continuation option;
  3. sql_warning_whenever : whenever_continuation option;
  4. sql_error_whenever : whenever_continuation option;
}
and whenever_continuation =
  1. | Continue
  2. | Perform of string
  3. | Goto of string
type declaration =
  1. | Simple_var_declaration of {
    1. prefix : string;
    2. var_level : string;

    4. var_type : string;
    5. var_content : string option;
    }
  2. | Field_var_declaration of {
    1. prefix : string;
    2. var_level : string;
    3. var_name : string;
    4. field : declaration list;
    }
type trans_stm =

    1. prefix : string;

    2. target : string;
    }

    1. prefix : string;
    2. condition : string;

    }
  6. | IfElse of {
    1. prefix : string;
    2. condition : string;

    1. prefix : string;

    1. content : string;

    }

    1. content : string;

    }
