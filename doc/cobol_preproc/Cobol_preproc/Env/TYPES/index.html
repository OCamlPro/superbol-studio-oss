type env = definition MAP.t

and definition = {
  1. def_loc : definition_loc;
  2. def_value : value;

  1. pp_payload : 'a;
  2. pp_loc : preproc_loc;
}

  1. | Source_location of Cobol_common.Srcloc.TYPES.srcloc
  2. | Process_parameter
  3. | Process_environment
and value =
  1. | Alphanum of Cobol_data.Value.alphanum with_preproc_loc
  2. | Boolean of Cobol_data.Value.boolean with_preproc_loc
  3. | Numeric of Cobol_data.Value.fixed with_preproc_loc

  1. prev_def_loc : definition_loc;
