    2. suppress_printing : bool;

  1. | CDirReplace of {
    1. also : bool;

    1. repl_subst : partial_subst Cobol_common.Srcloc.TYPES.with_loc;
    2. repl_to : string Cobol_common.Srcloc.TYPES.with_loc option;

}
